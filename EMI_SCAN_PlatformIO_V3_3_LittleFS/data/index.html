clear;
clc;
close all;

url = "http://192.168.4.1/data";

figure("Name","EMI SCAN - MATLAB");
tiledlayout(3,1);

ax1 = nexttile;
h1 = animatedline("LineWidth",1.5);
grid on;
title("Indice EMI relativo");
ylabel("EMI %");
ylim([0 100]);

ax2 = nexttile;
h2 = animatedline("LineWidth",1.5);
grid on;
title("E-Field / H-Field");
ylabel("%");
ylim([0 100]);
hold on;
h3 = animatedline("LineWidth",1.5);
legend("E-Field","H-Field");

ax3 = nexttile;
h4 = animatedline("LineWidth",1.5);
grid on;
title("Distancia ToF VL53L4CD");
ylabel("cm");
xlabel("Muestras");

n = 0;

while true
    n = n + 1;

    data = webread(url);

    emi = data.emiPercent;
    efield = data.ePercent;
    hfield = data.hPercent;

    if data.tof_valid == 1
        dist_cm = data.distance_cm;
    else
        dist_cm = NaN;
    end

    addpoints(h1,n,emi);
    addpoints(h2,n,efield);
    addpoints(h3,n,hfield);
    addpoints(h4,n,dist_cm);

    fprintf("EMI:%3.0f%% | E:%3.0f%% | H:%3.0f%% | Dist: %.1f cm | Mag: %.2f uT\n", ...
        emi, efield, hfield, dist_cm, data.mag_total);

    drawnow limitrate;
    pause(0.1);
end
